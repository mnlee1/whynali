### 2차: 8개 → 6개 (2026-03-13, 최종)

## DB 마이그레이션

### 실행 순서 (필수)

1. 백업
2. CHECK 제약 조건 삭제
3. 데이터 마이그레이션
4. 코드 배포
5. CHECK 제약 조건 재설정
6. 검증

### SQL 스크립트

```sql
-- 1. 백업
CREATE TABLE issues_backup AS SELECT * FROM issues;
CREATE TABLE news_data_backup AS SELECT * FROM news_data;

-- 2. CHECK 제약 조건 삭제
ALTER TABLE issues DROP CONSTRAINT IF EXISTS issues_category_check;
ALTER TABLE news_data DROP CONSTRAINT IF EXISTS news_data_category_check;

-- 3. 데이터 마이그레이션
UPDATE issues SET category = '기술' WHERE category = 'IT과학';
UPDATE news_data SET category = '기술' WHERE category = 'IT과학';

UPDATE issues SET category = '사회' WHERE category = '생활문화';
UPDATE news_data SET category = '사회' WHERE category = '생활문화';

-- 변경 결과 확인
SELECT category, COUNT(*) FROM issues GROUP BY category;
SELECT category, COUNT(*) FROM news_data GROUP BY category;

-- 4. CHECK 제약 조건 재설정
ALTER TABLE issues ADD CONSTRAINT issues_category_check 
    CHECK (category IN ('연예', '스포츠', '정치', '사회', '경제', '기술', '세계'));

ALTER TABLE news_data ADD CONSTRAINT news_data_category_check 
    CHECK (category IN ('연예', '스포츠', '정치', '사회', '경제', '기술', '세계'));

-- 5. 검증
SELECT * FROM issues 
WHERE category NOT IN ('연예', '스포츠', '정치', '사회', '경제', '기술', '세계');

SELECT * FROM news_data 
WHERE category NOT IN ('연예', '스포츠', '정치', '사회', '경제', '기술', '세계');
```
