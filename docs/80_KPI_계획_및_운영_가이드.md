```sql
-- [2] 일간 활동 추이 (최근 7일)
SELECT
    DATE(created_at) as 날짜,
    COUNT(*) as 댓글수
FROM comments
WHERE created_at >= NOW() - INTERVAL '7 days'
GROUP BY DATE(created_at)
ORDER BY 날짜 DESC;
```

```sql
-- [3] 참여율 계산
SELECT
    (SELECT COUNT(*) FROM auth.users WHERE deleted_at IS NULL) as 총가입자,
    (SELECT COUNT(DISTINCT user_id) FROM reactions) as 반응한_유저,
    ROUND(
        (SELECT COUNT(DISTINCT user_id) FROM reactions)::numeric /
        NULLIF((SELECT COUNT(*) FROM auth.users WHERE deleted_at IS NULL), 0) * 100,
        2
    ) as 반응참여율,
    (SELECT COUNT(DISTINCT user_id) FROM comments) as 댓글_작성_유저,
    ROUND(
        (SELECT COUNT(DISTINCT user_id) FROM comments)::numeric /
        NULLIF((SELECT COUNT(*) FROM auth.users WHERE deleted_at IS NULL), 0) * 100,
        2
    ) as 댓글참여율;
```

```sql
-- [4] 이슈당 평균 참여도
SELECT
    ROUND(AVG(reaction_count), 2) as 이슈당_평균_반응수,
    ROUND(AVG(comment_count), 2) as 이슈당_평균_댓글수
FROM (
    SELECT
        i.id,
        (SELECT COUNT(*) FROM reactions WHERE issue_id = i.id) as reaction_count,
        (SELECT COUNT(*) FROM comments WHERE issue_id = i.id) as comment_count
    FROM issues i
    WHERE approval_status = 'approved'
    AND is_hidden = false
) as issue_stats;
```

**방법 2: Supabase 로그 테이블 (무료)**

직접 조회수 테이블을 만들어 관리합니다.

```sql
-- 조회수 테이블 생성
CREATE TABLE page_views (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES auth.users(id),
    issue_id UUID REFERENCES issues(id),
    page_type TEXT NOT NULL, -- 'home', 'issue', 'discussion'
    created_at TIMESTAMPTZ DEFAULT NOW()
);

-- 인덱스 추가
CREATE INDEX idx_page_views_issue ON page_views(issue_id);
CREATE INDEX idx_page_views_created ON page_views(created_at);
CREATE INDEX idx_page_views_user ON page_views(user_id);

-- 일간 조회수 집계
SELECT
    DATE(created_at) as 날짜,
    COUNT(*) as 총조회수,
    COUNT(DISTINCT user_id) as 순방문자수
FROM page_views
WHERE created_at >= NOW() - INTERVAL '7 days'
GROUP BY DATE(created_at)
ORDER BY 날짜 DESC;
```

**권장: Google Analytics + Supabase 병행**
- Google Analytics: 전체 트래픽, DAU, 체류시간
- Supabase: 이슈별 조회수, 사용자별 활동 로그
